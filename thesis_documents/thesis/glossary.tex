\newglossaryentry{Dialog system/ dialog agent}{name={agent},
description={A computer system that can converse with human. In task-completion scenarios, the system aims to intelligently help the user achieve their goals (e.g. book a restaurant or rent a car) 
}}

\newglossaryentry{Dialog Strategy/ Dialog policy}{name={policy},
	description={Dialog policy dictates how the dialog agent should respond to user for a given state of the conversation.}}

\newglossaryentry{NLG (natural language generator)}{name={nlg},
	description={NLG is model which generates natural language speech from a provided semantic representation of speech goals. }}

\newglossaryentry{Semantic representation / semantic frame}{name={frame},
	description={ The semantic frame is a formalization in which the semantic meaning of an utterance is explicitly represented for machine comprehension.}}

\newglossaryentry{User Agenda}{name={agenda},
	description={ The user agenda is a stack like structure of pending dialogue acts, which serve as a convenient mechanisms for encoding the dialog’s history and user’s state of mind. }}


\newglossaryentry{NLU (natural language understanding)}{name={nlu},
	description={ NLU is a model which takes in a natural language sentence and converts it into a semantic representation that a computer system can ingest and act on.
}}
