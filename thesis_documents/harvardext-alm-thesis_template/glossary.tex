\newglossaryentry{allele}{name={Allele},
description={A name referring to a particular form of a polymorphism}}

\newglossaryentry{CallRate}{name={CallRate},
description={The fraction of successful data points}}

\newglossaryentry{DNA}{name={DNA},
description={DeoxyriboNucleic Acid, genetic material containing
instructions for the functioning of living organisms. They exist in
sequences of the four nucleotides \sym{A}, \sym{C}, \sym{G},
and \sym{T}}}

\newglossaryentry{gene}{name={Gene},
description={A functional stretch of DNA. Every protein is defined by a gene}}

\newglossaryentry{genome}{name={genome},
description={A complete DNA sequence of an organism}}

\newglossaryentry{geneexpression}{name={Gene Expression},
description={The quantification of the activity of a gene}}

\newglossaryentry{genomicfeature}{name={Genomic Feature},
description={An annotated stretch of DNA such as a gene or polymorphism}}

\newglossaryentry{genotype}{name={genotype},
description={The genetic makeup, or actual alleles, of a sample for a
specific polymorphism. For example, in a hypothetical family, the
mother has genotype \sym{C}/\sym{T} for rs25, the father \sym{C}/\sym{C}, and son \sym{C}/\sym{T}. The
allele \sym{C} of the son is inherited from the father and allele \sym{T} from the
mother}}

\newglossaryentry{pedigree}{name={Pedigree},
description={A structure showing relationship between members of an extended family}}

\newglossaryentry{phenotype}{name={phenotype}, 
description={A measurable or observable characteristic of a
sample. Gender is an example phenotype of string type that takes on
values of \sym{Female} and \sym{Male}. Weight is another example
of numeric type in pound unit}}


\newglossaryentry{polymorphism}{name={Polymorphism},
description={A name referring to anything that exists in multiple forms or variants}}

\newglossaryentry{SNP}{name={SNP},plural={SNPs},
description={Single Nucleotide Polymorphism, the most abundant type of DNA polymorphism that involves a single nucleotide and occurs at a particular genomic position. Each SNP is given a name like rs25. The majority of the SNPs have two expected alleles (e.g. rs25 has alleles C and T)}}

\newglossaryentry{sample}{name={sample},
description={A simple generic entity that represents a unique subject for which the user has data. It only contains a unique ID and a name}}
